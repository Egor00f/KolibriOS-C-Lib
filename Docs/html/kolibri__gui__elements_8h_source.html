187/* Add an element to an existing window */
188void kolibri_window_add_element(kolibri_window *some_window, enum KOLIBRI_GUI_ELEMENT_TYPE element_type, void *some_gui_element)
189{
190 kolibri_window_element *new_element = (kolibri_window_element *)malloc(sizeof(kolibri_window_element));
191
192 new_element -> type = element_type;
193 new_element -> element = some_gui_element;
194
195 if(!(some_window->elements)) /* No elements in window yet */
196 {
197 some_window->elements = new_element;
198 some_window->elements -> prev = some_window->elements;
199 some_window->elements -> next = some_window->elements;
200 }
201 else
202 {
203 kolibri_window_element *last_element = (kolibri_window_element*)some_window->elements->prev;
204
205 last_element -> next = new_element;
206 new_element -> next = some_window -> elements; /* start of linked list */
207 some_window -> elements -> prev = new_element;
208 new_element -> prev = last_element;
209 }
210}
