362 void text::TextLabel::Render() const
363 {
364 unsigned *FontSize = new unsigned;
365 *FontSize = 0;
366 if (_TextScale) // Если текст нужно подстраивать размер, то
367 { // Постраиваем
368 *FontSize = _size.x / lenghtPX();
369 if (!_Aligned)
370 {
371 for (int i = 0; i < _data.size(); i++)
372 {
373 Fonts::Font buff = _data[i].GetFont();
374 _data[i].SetFont(Fonts::Font(buff._Font, *FontSize, buff._Flags));
375 }
376 _Aligned = true;
377 }
378 }
379 delete FontSize;
380 Print(Coord(_coord.x + (_size.x / 2), _coord.y + (_size.y / 2)));
381 }
