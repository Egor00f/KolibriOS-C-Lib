33 unsigned GetFreeButtonId()
34 {
35 for (unsigned i = 0; i < ButtonsIdList.size(); i++) // Проходим по всему массиву
36 { // Если встречается свободный элемент,
37 if (!ButtonsIdList[i].use) // То используем его
38 { // Иначе создаём новый и использем тоже новый
