204 unsigned buttons::GetFreeButtonId()
205 {
206 for (unsigned i = 0; i < ButtonsIdList.size(); i++) // Проходим по всему массиву
207 { // Если встречается свободный элемент,
208 if (!ButtonsIdList[i]) // То используем его
209 { // Иначе создаём новый и использем тоже новый
210 ButtonsIdList[i] = true;
211 return i;
212 }
213 }
214
215 ButtonsIdList.push_back(true);
216 return ButtonsIdList.size() - 1; //-1 потому что обращение к элементам массива идёт с 0
