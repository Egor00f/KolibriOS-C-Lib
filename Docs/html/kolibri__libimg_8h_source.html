205inline Image_t* LoadImageFromFile(const char* Path)
206{
207 int32_t img_size;
208 FILE *f = fopen(Path, "rb"); // Этот код взят из примера из /contrib/C_Layer/EXAMPLE/img_example/main.c
209
210 if (!f)
211 {
212 const char a[] = "LoadImage: Can't open file: ";
213 char *b;
214 strcat(b, a);
215 strcat(b, Path);
216 strcat(b, " \n ");
217 _ksys_debug_puts(b);
218 }
219
220 if (fseek(f, 0, SEEK_END))
221 {
222 const char a[] = "Can't SEEK_END file: ";
223 char *b;
224 strcat(b, a);
225 strcat(b, Path);
226 strcat(b, " \n ");
227 _ksys_debug_puts(b);
228 }
229
230 int filesize = ftell(f);
231 rewind(f);
232 char *fdata = (char *)malloc(filesize);
233
234 if (!fdata)
235 {
236 const char a[] = "No memory for file: ";
237 char *b;
238 strcat(b, a);
239 strcat(b, Path);
240 strcat(b, "\n");
241 strcat(b, "malloc not return ptr\n");
242 _ksys_debug_puts(b);
243 }
244
245 img_size = (int32_t)fread(fdata, 1, filesize, f);
246
247 if (ferror(f))
248 {
249 const char *a = "Error reading file ";
250 char *b;
251 strcat(b, a);
252 strcat(b, Path);
253 strcat(b, " \n ");
254 _ksys_debug_puts(b);
255 }
256
257 fclose(f);
258
259 Image_t *buff = img_decode((void *)fdata, img_size, 0);
260
261 free(fdata);
262
263 return buff;
264}
