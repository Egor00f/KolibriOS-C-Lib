476 OS::Event Window::Handler()
477 {
478 OS::Event event = OS::WaitEvent(10);
479
480 switch (event)
481 {
482 case OS::Events::Redraw:
483
484 Render();
485
486 break;
487 case OS::Events::Button:
488
489 if(UI::buttons::GetPressedButton() == 1) //Если нажата кнопка X
490 {
491 return OS::Events::Exit;
492 }
493
494 for (unsigned i = 0; i < _Elements.size(); i++) //Запуск обработчиков всех используемых элементов
495 {
496 if (_Elements[i].use)
497 {
498 switch(_Elements[i]._type)
499 {
500 case Element::Type::Form:
501
502 _Elements[i].frm.ButtonHandler();
503 case Element::CheckBox:
504 _Elements[i].ChckBx.Handler();
505 case Element::Button:
506 _Elements[i].btn.Handler();
507 }
508 }
509 }
510
511 break;
512 case OS::Events::Key:
513
514 _Elements[activeForm].frm.Handler(); //Обработчик активной на текущий момент формы
515
516 break;
517 }
518 return event;
