225 unsigned Window::AddNewButton(UI::buttons::Button btn)
226 {

229 if (!_Elements[i].use)

231 switch (btn.GetType())
232 {
233 case UI::buttons::Content::Type::Image:
234 _Elements[i].btn.init(btn.GetCoord(), btn.GetSize(), btn.GetImage(), btn.GetMargin(), btn.GetColor());
235 break;
236 case UI::buttons::Content::Type::Text:
237 _Elements[i].btn.init(btn.GetCoord(), btn.GetSize(), btn.GetTextLabel(), btn.GetMargin(), btn.GetColor());

243 _Elements[i].use = true;
244 return i;

475 OS::Event Window::Handler()
476 {
477 OS::Event event = OS::WaitEvent(10); //ждём ивента 1/10 секунды
478
479 switch (event)
480 {
481 case OS::Events::Redraw:
482 Render();
483 break;
484 case OS::Events::Button:
485 if(UI::buttons::GetPressedButton() == 1)
486 {
487 return OS::Events::Exit;
488 }
489
490 for (unsigned i = 0; i < _Elements.size(); i++)
491 {
492 if (_Elements[i].use)
493 {
494 if (_Elements[i].type == Element::Type::Form)
495 {
496 _Elements[i].frm.ButtonHandler();
497 }
498 }
499 }
500 break;
501 case OS::Events::Key:
502 for(unsigned i = 0; i < _Elements.size(); i++)
503 {
504 if(_Elements[i].use)
505 {
506 if(_Elements[i].type == Element::Type::Form)
507 {
508 _Elements[activeForm].frm.Handler();
509 }
510 }
511 }
512 break;
513 default:
514 break;
515 }
516 return event;
517 }
