604 OS::Event Window::Handler()
605 {
606 OS::Event event = OS::WaitEvent(8);
607
608 switch (event)
609 {
610 case OS::Events::Redraw:
611
612 Render();
613
614 break;
615 case OS::Events::Button:
616
617 if (UI::buttons::GetPressedButton() == 1) // Если нажата кнопка X
618 {
619 return OS::Events::Exit;
620 }
621
622 for (const std::pair<int, Element> &n : _Elements) // Запуск обработчиков всех используемых элементов
623 {
624 switch (n.second._type)
625 {
626 case Element::Type::Form:
627 n.second.form->ButtonHandler();
628 case Element::CheckBox:
629 n.second.checkbox->Handler();
630 case Element::Button:
631 n.second.btn->Handler();
632 }
633 }
634
635 break;
636 case OS::Events::Key:
637
638 if (activeForm != -1)
639 {
640 _Elements[activeForm].form->Handler(); // Обработчик активной на текущий момент формы
641 }
642
643 break;
644 }
645 if (_RealtimeRedraw)
646 {
647 Coord Mouse = mouse::GetMousePositionInWindow();
648 if (((Mouse.x > 0 && Mouse.y > 0) && Mouse.x < GetWindowSize().x && Mouse.y < GetSkinHeight()) && mouse::GetMouseButtons() == mouse::LeftButton)
649 {
650 while (mouse::GetMouseButtons() == mouse::LeftButton)
651 {
652 Render(mouse::GetMousePositionOnSreen() - Mouse);
653 _ksys_thread_yield();
654 }
655 }
656 }
657
658 return event;
659 }
