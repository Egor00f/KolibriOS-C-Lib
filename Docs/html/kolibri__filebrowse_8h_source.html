231__attribute__((__stdcall__)) static inline void filebrowser_key(filebrowser *fb, oskey_t keypress)
233{
234// if (!fb->select_flag) return; // same reaction as other controls
235
236 int extended_key = 0, act = 0;
237
238 if(keypress.state) return;
239 if (keypress.code == 0xE0){ extended_key = 1; return; }
240
241 act = 0;
242 switch(keypress.ctrl_key) // ascii scancode
243 {
244 case 80: // arrow down
245 act = 1; break;
246 case 72: // arrow up
247 act = 2; break;
248 case 81: // PageDown
249 act = 3; break;
250 case 73: // PageUp
251 act = 4; break;
252 case 71: // Home
253 act = 5; break;
254 case 79: // End
255 act = 6; break;
256 case 28: // Enter
257 act = 7; break;
258 case 82: // Insert
259 act = 8; break;
260 case 78: // NumPad+ select all
261 act = 9; break;
262 case 74: // NumPad- deselct
263 act = 10; break;
264 case 55: // NumPad* invert selection
265 act = 11; break;
266 default:
267 act = 12; // search by letter
268 }
269 fb->key_action = act;
270 fb->key_action_num = keypress.ctrl_key;
271
272// debug_board_printf("key pressed [%X] %d, action %d, ext_flag = %d\n", keypress.val, brows.key_action_num, act, extended_key);
273
274 if (extended_key) extended_key = 0;
275 (*filebrowse_key)(fb);
276}
